    types.hpp

    init.hpp rng.hpp
    level_gen.hpp level_gen.cpp
)

    ${SIMULATOR_SRCS}
)

    PUBLIC
        madrona_mw_core

        nlohmann_json::nlohmann_json
        madrona_common
        madrona_mw_physics
        madrona_viz_system
)

    mgr.hpp mgr.cpp

        madrona_common
